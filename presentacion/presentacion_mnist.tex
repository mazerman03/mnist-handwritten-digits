%========== SLIDE 4: MODELO ==========
\begin{frame}{Modelo: Red Neuronal MLP}
\begin{columns}
\column{0.5\textwidth}
\textbf{Arquitectura:}
\begin{center}
\begin{tabular}{ll}
\toprule
\textbf{Capa} & \textbf{Neuronas} \\
\midrule
Entrada & 784 (28$\times$28 px) \\
Oculta 1 & 128 (ReLU) \\
Oculta 2 & 64 (ReLU) \\
Salida & 10 (dígitos 0--9) \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.3cm}
\textbf{Hiperparámetros:}
\begin{itemize}
    \item Optimizador: Adam
    \item Learning rate: 0.001
    \item Batch size: 256
\end{itemize}

\column{0.5\textwidth}
\textbf{Preprocesamiento:}
\begin{itemize}
    \item Normalización de píxeles a [0, 1]
    \item División estándar 60k/10k
\end{itemize}

\vspace{0.3cm}
\textbf{¿Qué es un MLP?}
\begin{itemize}
    \item Red neuronal con capas \textbf{completamente conectadas}
    \item Cada neurona aplica una suma ponderada + función de activación
    \item Aprende mediante \textbf{backpropagation}
\end{itemize}
\end{columns}
\end{frame}

\textbf{Creación:}
\begin{enumerate}
    \item Dibujar dígitos con marcador en papel blanco
    \item Fotografiar cada dígito
    \item Preprocesar automáticamente:
    \begin{itemize}
        \item Convertir a escala de grises
        \item Binarizar y recortar al dígito
        \item Redimensionar a 28$\times$28
        \item Centrar con padding (como MNIST)
        \item Invertir colores (fondo negro)
    \end{itemize}
\end{enumerate}

%========== SLIDE 10: CONCLUSIONES ==========
\begin{frame}{Conclusiones}
\begin{enumerate}
    \item Un \textbf{MLP sencillo} (2 capas ocultas) logra $>$97\% accuracy en MNIST
    \item El rendimiento en imágenes propias depende fuertemente de la \textbf{calidad del preprocesamiento}
    \item Los dígitos más difíciles comparten \textbf{rasgos visuales similares} (ej: 3/5, 4/9, 1/7)
    \item \textbf{Lección clave:} un modelo es tan bueno como la representatividad de sus datos de entrenamiento respecto a los datos reales
\end{enumerate}

\vspace{0.5cm}
\begin{block}{Posibles mejoras}
\begin{itemize}
    \item Usar redes convolucionales (CNN) para capturar mejor los patrones espaciales
    \item Aumentar datos de entrenamiento con transformaciones (data augmentation)
    \item Ajustar hiperparámetros con validación cruzada
\end{itemize}
\end{block}
